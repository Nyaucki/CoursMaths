\section{Décomposition en produit de facteurs premiers}

\prop{Décomposition en produit de facteur premier}
{Tout nombre entier peut être décomposé en une multiplication de nombres premiers.}

\exmpl{
    \begin{itemize}
        \item $60=2\times2\times3\times5$
        \item $90=2\times3\times3\times5$
    \end{itemize}
}

\rmq{Cette décomposition est utile pour simplifier des fractions. }

\exmpl{$\dfrac{60}{90}=\dfrac{2\times2\times3\times5}{2\times3\times3\times5}=\dfrac{\cancel{2}\times2\times\cancel{3}\times\cancel{5}}{\cancel{2}\times3\times\cancel{3}\times\cancel{5}}=\dfrac{2}{3}$}
