\section{Réduire une expression littérale}

\prop{Factorisation}
{Pour n'importe quels nombres $a,b$ et $c$ :
$$ac+bc=(a+b)c$$}

\exmpl{
    \begin{itemize}
        \item $3x+5x=8x$
        \item $7x-12x=-5x$
        \item $3x+4y$ ne peut pas être modifié
        \item $7x-3$ ne peut pas être modifié
    \end{itemize}
}

\rmq{On peut donc uniquement faire des opérations avec les termes ayant la même lettre. Quand on le fait pour réduire le nombre de termes, on dit qu'on réduit l'expression.}

\prop{Distributivité}
{Pour n'importe quels nombres $a,b$ et $c$ :
$$a(b+c)=ab+ac$$}

\exmpl{
    \begin{itemize}
        \item $4(3x+2)=4\times3x+4\times2=12x+8$
        \item $5(7x-3)=5\times7x-5\times3$
    \end{itemize}
}
