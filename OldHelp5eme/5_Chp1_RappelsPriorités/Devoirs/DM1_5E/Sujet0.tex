			&\py{a1}+\py{b1}\times\py{c1}\\

			=&\py{a1+b1*c1}\\
		\end{align*}

		\item 
		\begin{align*}
			&\py{a2}-\py{b2}+\py{c2}\times\py{d2}\\
			=&\py{a2}-\py{b2}+\py{c2*d2}\\
			=&\py{a2-b2}+\py{c2*d2}\\
			=&\py{a2-b2+c2*d2}\\
		\end{align*}
	\end{multicols}
	\begin{multicols}{2}
		\item 
		\begin{align*}
			&(\py{a3}+\py{b3})\times\py{c3}-\py{d3}\\
			=&\py{a3+b3}\times\py{c3}-\py{d3}\\
			=&\py{(a3+b3)*c3}-\py{d3}\\
			=&\py{(a3+b3)*c3-d3}\\
		\end{align*}

		\item 
		\begin{align*}
			&\py{a4}\times  (\py{b4}+\py{c4}\times (\py{d4}-\py{e4}))\\
			=&\py{a4}\times  (\py{b4}+\py{c4}\times \py{d4-e4})\\
			=&\py{a4}\times  (\py{b4}+\py{c4*(d4-e4)})\\		
			=&\py{a4}\times  \py{b4+c4*(d4-e4)}\\		
			=&\py{a4*(b4+c4*(d4-e4))}\\		
		\end{align*}
