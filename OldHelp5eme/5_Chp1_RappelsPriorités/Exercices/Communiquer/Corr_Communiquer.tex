\begin{multicols}{2}
  \exo{Communiquer}{TrouverErreur1}
  Trouve l'erreur :
  \begin{align*}
      &12:3+1\times 3&&\\
      =&12:\textbf{3+3}&&\\
      =&12:\textbf{6}&&\text{Addition avant division}\\
      =&2
  \end{align*}

  \exo{Communiquer}{TrouverErreur2}
  Trouve l'erreur :
  \begin{align*}
      &\textbf{2+8}:4-2&&\\
      =&\textbf{10}:4-2&&\text{Addition avant division}\\
      =&2,5-2&&\\
      =&0,5
  \end{align*}  
\end{multicols}

  \exo{Communiquer}{TrouverErreur3}
  Trouve l'erreur :
  \begin{align*}
      &(\textbf{5+1}\times 2)\times 4+1&&\\
      =&(\textbf{6}\times 2)\times 4+1&&\text{Addition avant produit}\\
      =&12\times 4+1&&\\
      =&48+1&&\\
      =&49
  \end{align*}

  \exo{Communiquer}{TrouverErreur4}
  Trouve l'erreur :
  \begin{align*}
      &14-3+2\times (5-1)&&\\
      =&14-3+2\times 4&&\\
      =&14-\textbf{3+8}&&\\
      =&14-\textbf{11}&&\text{Pas de gauche à droite}\\
      =&3
  \end{align*}
\end{multicols}

      =&28:\textbf{4+3}\times (12-7)&&\\
