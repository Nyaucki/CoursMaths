\begin{multicols}{2}
    \exo{Communiquer}{TrouverErreur1}
    Trouve l'erreur :
    \begin{align*}
        &12:3+1\times 3\\
        =&12:3+3\\
        =&12:6\\
        =&2
    \end{align*}

    \exo{Communiquer}{TrouverErreur2}
    Trouve l'erreur :
    \begin{align*}
        &2+8:4-2\\
        =&10:4-2\\
        =&2,5-2\\
        =&0,5
    \end{align*}  
\end{multicols}

    \exo{Communiquer}{TrouverErreur3}
    Trouve l'erreur :
    \begin{align*}
        &(5+1\times 2)\times 4+1\\
        =&(6\times 2)\times 4+1\\
        =&12\times 4+1\\
        =&48+1\\
        =&49
    \end{align*}

        =&14-3+2\times 4\\
        =&14-3+8\\
        =&14-11\\
        =&3
    \end{align*}
\end{multicols}
