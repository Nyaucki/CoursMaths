\section{Nombres premiers}

\dfnt{Nombres premiers}
{Un nombre premier est un nombre entier possédant exactement deux diviseurs : 1 et lui-même.}

\rmq{La liste des nombres premiers commence par : 2 ; 3 ; 5 ; 7 ; 11 ; 13 ; 19 ; 23 ; 29}

\prop{Décomposition en produit de facteurs premiers}
{Tout nombre supérieur à 2 se décompose de manière unique en produit de facteurs premiers.}

\rmq{La décomposition en produit de facteurs premiers permet de simplifier des fractions.}
