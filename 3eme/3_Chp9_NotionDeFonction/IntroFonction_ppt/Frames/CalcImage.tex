\begin{frame}
    \underbar{On considère la fonction $G:x\mapsto -x^2+4x-3$}\vspace*{1em}

    \begin{enumerate}
        \item Donner le programme associé  à la fonction $G$.
        \item Qu'obtient avec pour nombre de départs 3 ?
    \end{enumerate}
    \vfill
\end{frame}

\begin{frame}
    \underbar{On considère la fonction $I:x\mapsto -x^2+4x-3$}\vspace*{1em}

    \begin{enumerate}
        \item Donner le programme associé  à la fonction $I$.
        \item Qu'obtient avec pour nombre de départs 3 ?
    \end{enumerate}
    \vspace*{1em}

    On peut écrire $I(3)=0$, et on dit que 0 est l'image de 3 par la fonction $I$.
    \vfill
\end{frame}

        \item Donner l'image de $4$ par la fonction $J$
    \end{enumerate}
\end{frame}
