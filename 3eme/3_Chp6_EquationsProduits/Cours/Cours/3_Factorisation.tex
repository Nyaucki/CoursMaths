\section{Factorisation}



\prop{Identité remarquable}{
    Pour tous réels $a \text{ et } b$, on a :
    \begin{equation*}
        a^2-b^2=(a+b)(a-b)%au tableau, mettre des couleur pour expliquer
    \end{equation*}
}

\exmpl{$4x^2-16=(2x-4)(2x+4)$}

\prop{Facteur commun}{
Pour tous réels $k,a \text{ et } b$, on a :
\begin{align*}
    k\times a + k \times b=k\times(a+b)\\
    k\times a - k \times b= k\times(a-b)
\end{align*}
On dit alors que $k$ est un \textbf{facteur commun}.
}

\rmq{Le facteur commun peut être une expression}

\exmpl{$(4x-3)(3x+4)-(4x-3)(5x-4)=(4x-3)((3x+4)-(5x-4))$}
