\exo{Représenter}{HistoToTab1} 

\begin{center} 
\begin{tikzpicture}
\draw [<->] (0,5.5) node[above left]{Effectifs}|- (4.5,0) node[below right]{Valeurs};
\foreach \x in {0,...,5}
{
\draw[lightgray,thin] (0,\x)--(4.5,\x);
}
\foreach \z [count=\zi] in {7, 10, 13, 15}{
\node at (\zi - 0.5,-0.5) {\z};
}
\foreach \x in {0,...,5}
{
\draw (-0.1,\x) node[left]{\x} --(0.1,\x);
}
\foreach \y [count=\yi] in {5, 2, 4, 1}
{
\draw[fill=white] (\yi -1,0) rectangle (\yi,\y);
}
\end{tikzpicture}
\end{center}
