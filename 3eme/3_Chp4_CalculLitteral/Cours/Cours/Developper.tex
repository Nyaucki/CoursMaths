\section{Développer et réduire}

\prop{Simple distributivité}{
    Pour tous réels $k,a \text{ et } b$, on a :
    \begin{align*}
        k\times(a+b)=k\times a + k \times b \\
        k\times(a-b)=k\times a - k \times b
    \end{align*}
}

\exmpl{
\begin{itemize}
    \item $3(x+1)=3 \times x + 3\times 1 =3x+3$
    \item $x(2x-4)=2x^2-4x$
    \item $-2(100-40x)=-200+80x$
\end{itemize}
}

\rmq{ Développer, c'est transformer un produit en une somme. }
    

\prop{Double distributivité}{
    Pour tous réels $a,b,c \text{ et } d$, on a :
    \begin{equation*}
        (a+b)(c+d)=ac+ad+bc+bd %au tableau, mettre des couleur pour expliquer
    \end{equation*}
}

\exmpl{
\begin{itemize}
    \item $(2x+3)(4x-1)=2x\times 4x + 2x \times (-1) + 3 \times 4x + 3 \times (-1)=8x^2+10x-3$
    \item $(3x-1)(-2x-5)=3x\times (-2x)+3x\times (-5)-1\times (-2x)-1\times (-5)=-6x^2-13x+5$
    \item $(x+1)(2x^2-4)=x\times 2x^2+x\times (-4)+1\times 2x^2+1\times (-4)=2x^3+2x^2-4x-4$
\end{itemize}
}
